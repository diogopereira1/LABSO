2-Usando estrutura de repetição, faça um script (seunome2.sh) para calcular e exibir quantas
pastas e quantos arquivos tem na pasta atual.


nano diogoramalho2.sh



#!/bin/bash

pastas=0

arquivos=0


for item in *

do

 if [ -d $item ]
 
 then
  
  pastas=$(($pastas+1))
 
 else
  
  arquivos=$(($arquivos+1))
 
 fi

done

echo "Arquivos: $arquivos diretorios: $pastas"
